1 /++

4 This module provides [SumType], an alternative to `std.variant.Algebraic` with
5 [match|improved pattern-matching], full attribute correctness (`pure`, `@safe`,

14 /// $(H3 Basic usage)
15 @safe unittest {
16     import std.math: approxEqual;
17 
18     struct Fahrenheit { double degrees; }
19     struct Celsius { double degrees; }
20     struct Kelvin { double degrees; }
21 
22     alias Temperature = SumType!(Fahrenheit, Celsius, Kelvin);
23 
24     pure @safe @nogc nothrow
25     Fahrenheit toFahrenheit(Temperature t)
26     {
27         return Fahrenheit(
28             t.match!(
29                 (Fahrenheit f) => f.degrees,
30                 (Celsius c) => c.degrees * 9.0/5 + 32,
31                 (Kelvin k) => k.degrees * 9.0/5 - 459.4
32             )
33         );
34     }
35 
36     Temperature t1 = Fahrenheit(98.6);
37     Temperature t2 = Celsius(100);
38     Temperature t3 = Kelvin(273);
39 
40     assert(toFahrenheit(t1).degrees.approxEqual(98.6));
41     assert(toFahrenheit(t2).degrees.approxEqual(212));
42     assert(toFahrenheit(t3).degrees.approxEqual(32));
43 }

45 /** $(H3 Structural matching)
46  *
47  * In the `length` and `horiz` functions below, the handlers for `match` do not
48  * specify the types of their arguments. Instead, matching is done based on how
49  * the argument is used in the body of the handler: any type with `x` and `y`
50  * properties will be matched by the `rect` handlers, and any type with `r` and
51  * `theta` properties will be matched by the `polar` handlers.
52  */
53 @safe unittest {
54     import std.math: approxEqual, cos, PI, sqrt;
55 
56     struct Rectangular { double x, y; }
57     struct Polar { double r, theta; }
58     alias Vector = SumType!(Rectangular, Polar);
59 
60     pure @safe @nogc nothrow
61     double length(Vector v)
62     {
63         return v.match!(
64             rect => sqrt(rect.x^^2 + rect.y^^2),
65             polar => polar.r
66         );
67     }
68 
69     pure @safe @nogc nothrow
70     double horiz(Vector v)
71     {
72         return v.match!(
73             rect => rect.x,
74             polar => polar.r * cos(polar.theta)
75         );
76     }
77 
78     Vector u = Rectangular(1, 1);
79     Vector v = Polar(1, PI/4);
80 
81     assert(length(u).approxEqual(sqrt(2.0)));
82     assert(length(v).approxEqual(1));
83     assert(horiz(u).approxEqual(1));
84     assert(horiz(v).approxEqual(sqrt(0.5)));
85 }

87 /** $(H3 Arithmetic expression evaluator)
88  *
89  * This example makes use of the special placeholder type `This` to define a
90  * [https://en.wikipedia.org/wiki/Recursive_data_type|recursive data type]: an
91  * [https://en.wikipedia.org/wiki/Abstract_syntax_tree|abstract syntax tree] for
92  * representing simple arithmetic expressions.
93  */
94 @safe unittest {
95     import std.functional: partial;
96     import std.traits: EnumMembers;
97     import std.typecons: Tuple;
98 
99     enum Op : string
100     {
101         Plus  = "+",
102         Minus = "-",
103         Times = "*",
104         Div   = "/"
105     }
106 
107     // An expression is either
108     //  - a number,
109     //  - a variable, or
110     //  - a binary operation combining two sub-expressions.
111     alias Expr = SumType!(
112         double,
113         string,
114         Tuple!(Op, "op", This*, "lhs", This*, "rhs")
115     );
116 
117     // Shorthand for the Tuple type above
118     alias BinOp = Expr.Types[2];
119 
120     // Factory function for number expressions
121     pure @safe
122     Expr* num(double value)
123     {
124         return new Expr(value);
125     }
126 
127     // Factory function for variable expressions
128     pure @safe
129     Expr* var(string name)
130     {
131         return new Expr(name);
132     }
133 
134     // Factory function for binary operation expressions
135     pure @safe
136     Expr* binOp(Op op, Expr* lhs, Expr* rhs)
137     {
138         return new Expr(BinOp(op, lhs, rhs));
139     }
140 
141     // Convenience wrappers for creating BinOp expressions
142     alias sum  = partial!(binOp, Op.Plus);
143     alias diff = partial!(binOp, Op.Minus);
144     alias prod = partial!(binOp, Op.Times);
145     alias quot = partial!(binOp, Op.Div);
146 
147     // Evaluate expr, looking up variables in env
148     pure @safe nothrow
149     double eval(Expr expr, double[string] env)
150     {
151         return expr.match!(
152             (double num) => num,
153             (string var) => env[var],
154             (BinOp bop) {
155                 double lhs = eval(*bop.lhs, env);
156                 double rhs = eval(*bop.rhs, env);
157                 final switch(bop.op) {
158                     static foreach(op; EnumMembers!Op) {
159                         case op:
160                             return mixin("lhs" ~ op ~ "rhs");
161                     }
162                 }
163             }
164         );
165     }
166 
167     // Return a "pretty-printed" representation of expr
168     @safe
169     string pprint(Expr expr)
170     {
171         import std.format;
172 
173         return expr.match!(
174             (double num) => "%g".format(num),
175             (string var) => var,
176             (BinOp bop) => "(%s %s %s)".format(
177                 pprint(*bop.lhs),
178                 bop.op,
179                 pprint(*bop.rhs)
180             )
181         );
182     }
183 
184     Expr* myExpr = sum(var("a"), prod(num(2), var("b")));
185     double[string] myEnv = ["a":3, "b":4, "c":7];
186 
187     assert(eval(*myExpr, myEnv) == 11);
188     assert(pprint(*myExpr) == "(a + (2 * b))");
189 }

195 /**
196  * A tagged union that can hold a single value from any of a specified set of
197  * types.
198  *
199  * The value in a `SumType` can be operated on using [match|pattern matching].
200  *
201  * The special type `This` can be used as a placeholder to create
202  * self-referential types, just like with `Algebraic`. See the
203  * [sumtype#arithmetic-expression-evaluator|"Arithmetic expression evaluator" example] for
204  * usage.
205  *
206  * A `SumType` is initialized by default to hold the `.init` value of its
207  * first member type, just like a regular union. The version identifier
208  * `SumTypeNoDefaultCtor` can be used to disable this behavior.
209  *
210  * To avoid ambiguity, duplicate types are not allowed (but see the
211  * [sumtype#basic-usage|"basic usage" example] for a workaround).
212  *
213  * Bugs:
214  *   Types with non-`const` overloads of `opEquals` cannot be stored in a
215  *   `SumType`.
216  *
217  * See_Also: `std.variant.Algebraic`
218  */
219 struct SumType(TypeArgs...)
220 	if (is(NoDuplicates!TypeArgs == TypeArgs)
221 	    && TypeArgs.length > 0
222 	    && TypeArgs.length < size_t.max)
223 {
224 	import std.meta: AliasSeq;
225 	import std.typecons: ReplaceType;
226 
227 	/// The types a `SumType` can hold.
228 	alias Types = AliasSeq!(ReplaceType!(This, typeof(this), TypeArgs));
229 
230 private:
231 
232 	import std.meta: AliasSeq, Filter;
233 
234 	enum bool canHoldTag(T) = Types.length <= T.max;
235 	alias unsignedInts = AliasSeq!(ubyte, ushort, uint, ulong);
236 
237 	alias Tag = Filter!(canHoldTag, unsignedInts)[0];
238 
239 	union Storage
240 	{
241 		Types values;
242 
243 		static foreach (i, T; Types) {
244 			@trusted
245 			this(inout(T) val) inout
246 			{
247 				values[i] = val;
248 			}
249 		}
250 	}
251 
252 	Tag tag;
253 	Storage storage;
254 
255 	@trusted
256 	ref inout(T) trustedGet(T)() inout
257 	{
258 		import std.meta: staticIndexOf;
259 
260 		enum tid = staticIndexOf!(T, Types);
261 		assert(tag == tid);
262 		return storage.values[tid];
263 	}
264 
265 public:
266 
267 	static foreach (i, T; Types) {
268 		/// Constructs a `SumType` holding a specific value.
269 		this(inout(T) val) inout
270 		{
271 			tag = i;
272 			storage = inout(Storage)(val);
273 		}
274 	}
275 
276 	version(SumTypeNoDefaultCtor) {
277 		@disable this();
278 	}
279 
280 	import std.traits: isAssignable;
281 
282 	static foreach (i, T; Types) {
283 		static if (isAssignable!T) {
284 			/// Assigns a value to a `SumType`.
285 			void opAssign(T rhs)
286 			{
287 				import std.traits: hasElaborateDestructor;
288 
289 				this.match!((ref value) {
290 					static if (hasElaborateDestructor!(typeof(value))) {
291 						destroy(value);
292 					}
293 				});
294 
295 				tag = i;
296 				storage = Storage(rhs);
297 			}
298 		}
299 	}
300 
301 	import std.meta: allSatisfy;
302 	import std.traits: isEqualityComparable;
303 
304 	static if (allSatisfy!(isEqualityComparable, Types)) {
305 		/**
306 		 * Compares two `SumType`s for equality.
307 		 *
308 		 * Two `SumType`s are equal if they contain values of the same type,
309 		 * and those values are equal.
310 		 *
311 		 * Bugs:
312 		 *   Comparison of `SumType`s that contain a `const(This)[]`, either
313 		 *   directly or indirectly, will sometimes give the wrong result by
314 		 *   comparing the arrays for reference equality (`is`) instead of
315 		 *   value equality (`==`).
316 		 */
317 		bool opEquals(const SumType rhs) const {
318 			return this.match!((ref value) {
319 				return rhs.match!((ref rhsValue) {
320 					static if (is(typeof(value) == typeof(rhsValue))) {
321 						return value == rhsValue;
322 					} else {
323 						return false;
324 					}
325 				});
326 			});
327 		}
328 	} else {
329 		// Don't fall back to default (bitwise) equality
330 		@disable bool opEquals(const SumType rhs) const;
331 	}
332 
333 	import std.meta: anySatisfy;
334 	import std.traits: hasElaborateCopyConstructor, hasElaborateDestructor;
335 
336 	// Workaround for dlang issue 19407
337 	static if (__traits(compiles, anySatisfy!(hasElaborateDestructor, Types))) {
338 		// If possible, include the destructor only when it's needed
339 		private enum includeDtor = anySatisfy!(hasElaborateDestructor, Types);
340 	} else {
341 		// If we can't tell, always include it, even when it does nothing
342 		private enum includeDtor = true;
343 	}
344 
345 	static if (includeDtor) {
346 		/// Calls the destructor of the `SumType`'s current value.
347 		~this()
348 		{
349 			this.match!((ref value) {
350 				static if (hasElaborateDestructor!(typeof(value))) {
351 					destroy(value);
352 				}
353 			});
354 		}
355 	}
356 
357 	static if (anySatisfy!(hasElaborateCopyConstructor, Types)) {
358 		/// Calls the postblit of the `SumType`'s current value.
359 		this(this)
360 		{
361 			this.match!((ref value) {
362 				static if (hasElaborateCopyConstructor!(typeof(value))) {
363 					value.__xpostblit;
364 				}
365 			});
366 		}
367 	}
368 
369 	/// Returns a string representation of a `SumType`'s value.
370 	string toString() const {
371 		import std.conv: text;
372 		return this.match!((auto ref value) {
373 			return value.text;
374 		});
375 	}
376 }

649 /**
650  * Calls a type-appropriate function with the value held in a [SumType].
651  *
652  * For each possible type the [SumType] can hold, the given handlers are
653  * checked, in order, to see whether they accept a single argument of that type.
654  * The first one that does is chosen as the match for that type.
655  *
656  * Implicit conversions are not taken into account, except between
657  * differently-qualified versions of the same type. For example, a handler that
658  * accepts a `long` will not match the type `int`, but a handler that accepts a
659  * `const(int)[]` will match the type `immutable(int)[]`.
660  *
661  * Every type must have a matching handler, and every handler must match at
662  * least one type. This is enforced at compile time.
663  *
664  * Handlers may be functions, delegates, or objects with opCall overloads. If a
665  * function with more than one overload is given as a handler, all of the
666  * overloads are considered as potential matches.
667  *
668  * Templated handlers are also accepted, and will match any type for which they
669  * can be [implicitly instantiated](https://dlang.org/glossary.html#ifti). See
670  * [sumtype#structural-matching|"Structural matching"] for an example of
671  * templated handler usage.
672  *
673  * Returns:
674  *   The value returned from the handler that matches the currently-held type.
675  *
676  * See_Also: `std.variant.visit`
677  */
678 template match(handlers...)
679 {
680 	import std.typecons: Yes;
681 
682 	/**
683 	 * The actual `match` function.
684 	 *
685 	 * Params:
686 	 *   self = A [SumType] object
687 	 */
688 	auto match(Self)(auto ref Self self)
689 		if (is(Self : SumType!TypeArgs, TypeArgs...))
690 	{
691 		return self.matchImpl!(Yes.exhaustive, handlers);
692 	}
693 }
694 
695 /**
696  * Attempts to call a type-appropriate function with the value held in a
697  * [SumType], and throws on failure.
698  *
699  * Matches are chosen using the same rules as [match], but are not required to
700  * be exhaustive—in other words, a type is allowed to have no matching handler.
701  * If a type without a handler is encountered at runtime, a [MatchException]
702  * is thrown.
703  *
704  * Returns:
705  *   The value returned from the handler that matches the currently-held type,
706  *   if a handler was given for that type.
707  *
708  * Throws:
709  *   [MatchException], if the currently-held type has no matching handler.
710  *
711  * See_Also: `std.variant.tryVisit`
712  */
713 template tryMatch(handlers...)
714 {
715 	import std.typecons: No;
716 
717 	/**
718 	 * The actual `tryMatch` function.
719 	 *
720 	 * Params:
721 	 *   self = A [SumType] object
722 	 */
723 	auto tryMatch(Self)(auto ref Self self)
724 		if (is(Self : SumType!TypeArgs, TypeArgs...))
725 	{
726 		return self.matchImpl!(No.exhaustive, handlers);
727 	}
728 }

732 {
733 	pure @safe @nogc nothrow
734 	this(string msg, string file = __FILE__, size_t line = __LINE__)
735 	{
736 		super(msg, file, line);
737 	}

740 /**
741  * Checks whether a handler can match a given type.
742  *
743  * See the documentation for [match] for a full explanation of how matches are
744  * chosen.
745  */
746 template canMatch(alias handler, T)
747 {
748 	private bool canMatchImpl()
749 	{
750 		import std.traits: hasMember, isCallable, isSomeFunction, Parameters;
751 
752 		// immutable recursively overrides all other qualifiers, so the
753 		// right-hand side is true if and only if the two types are the
754 		// same up to qualifiers (i.e., they have the same structure).
755 		enum sameUpToQuals(T, U) = is(immutable(T) == immutable(U));
756 
757 		bool result = false;
758 
759 		static if (is(typeof({ T dummy = T.init; handler(dummy); }))) {
760 			// Regular handlers
761 			static if (isCallable!handler) {
762 				// Functions and delegates
763 				static if (isSomeFunction!handler) {
764 					static if (sameUpToQuals!(T, Parameters!handler[0])) {
765 						result = true;
766 					}
767 				// Objects with overloaded opCall
768 				} else static if (hasMember!(typeof(handler), "opCall")) {
769 					static foreach (overload; __traits(getOverloads, typeof(handler), "opCall")) {
770 						static if (sameUpToQuals!(T, Parameters!overload[0])) {
771 							result = true;
772 						}
773 					}
774 				}
775 			// Generic handlers
776 			} else {

833 private template matchImpl(Flag!"exhaustive" exhaustive, handlers...)
834 {
835 	auto matchImpl(Self)(auto ref Self self)
836 		if (is(Self : SumType!TypeArgs, TypeArgs...))
837 	{
838 		import std.meta: staticMap;
839 
840 		alias Types = self.Types;
841 		enum noMatch = size_t.max;
842 
843 		alias allHandlers = staticMap!(overloadInclusiveHandler, handlers);
844 
845 		pure size_t[Types.length] getHandlerIndices()
846 		{
847 			size_t[Types.length] indices;
848 			indices[] = noMatch;
849 
850 			static foreach (tid, T; Types) {
851 				static foreach (hid, handler; allHandlers) {
852 					static if (canMatch!(handler, typeof(self.trustedGet!T()))) {
853 						if (indices[tid] == noMatch) {
854 							indices[tid] = hid;
855 						}
856 					}
857 				}
858 			}
859 
860 			return indices;
861 		}
862 
863 		enum handlerIndices = getHandlerIndices;
864 
865 		final switch (self.tag) {
866 			static foreach (tid, T; Types) {
867 				case tid:
868 					static if (handlerIndices[tid] != noMatch) {
869 						return allHandlers[handlerIndices[tid]](self.trustedGet!T);
870 					} else {
871 						static if(exhaustive) {
872 							static assert(false,
873 								"No matching handler for type `" ~ T.stringof ~ "`");
874 						} else {
875 							throw new MatchException(
876 								"No matching handler for type `" ~ T.stringof ~ "`");
877 						}
878 					}
879 			}
880 		}
881 
882 		assert(false); // unreached
883 
884 		import std.algorithm.searching: canFind;
885 
886 		// Check for unreachable handlers
887 		static foreach (hid, handler; allHandlers) {
888 			static assert(handlerIndices[].canFind(hid),
889 				"handler `" ~ __traits(identifier, handler) ~ "` " ~
890 				"of type `" ~ typeof(handler).stringof ~ "` " ~
891 				"never matches"
892 			);
893 		}
894 	}
895 }
